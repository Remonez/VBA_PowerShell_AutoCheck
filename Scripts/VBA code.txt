Sub CheckExcelFile()
    Dim targetDate As Date
    Dim lastRow As Long
    Dim currentRow As Long
    Dim conditionMet As Boolean
    Dim rowsAfterTomorrow As String
    
    ' Set the target date for the day after tomorrow
    targetDate = Date + 2 ' Change to +1 for tomorrow

    ' Initialize the conditionMet variable
    conditionMet = False
    
    ' Initialize the variable to store rows with dates after tomorrow
    rowsAfterTomorrow = ""

    ' Find the last row in column B
    lastRow = Cells(Rows.Count, 2).End(xlUp).Row

    ' Loop through each row in column B
    For currentRow = 1 To lastRow
        ' Get the date from the current row of column B
        Dim currentDate As Variant
        currentDate = Cells(currentRow, 2).Value

        ' Check if the value in column B is a valid date
        If IsDate(currentDate) Then
            ' Check if the date in column B is after tomorrow
            If currentDate = targetDate Then
                ' Add the current row to the list of rows after tomorrow
                rowsAfterTomorrow = rowsAfterTomorrow & "Row " & currentRow & vbCrLf
                
                ' Set the conditionMet variable to True
                conditionMet = True
            End If
        End If
    Next currentRow

    ' Check the conditionMet variable
    If conditionMet Then
        ' Display a message with all rows after tomorrow
        MsgBox "The following dates are after tomorrow:" & vbCrLf & rowsAfterTomorrow, vbInformation, "Date Check"
    Else
        ' If the condition wasn't met, you can close Excel here
        Application.Quit
    End If
End Sub
