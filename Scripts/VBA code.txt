Sub CheckExcelFile()
    Dim targetDateTomorrow As Date
    Dim targetDateDayAfterTomorrow As Date
    Dim lastRow As Long
    Dim currentRow As Long
    Dim tomorrowRows As String
    Dim afterTomorrowRows As String
    Dim conditionMet As Boolean

    ' Set the target dates for tomorrow and the day after tomorrow
    targetDateTomorrow = Date + 1
    targetDateDayAfterTomorrow = Date + 2

    ' Initialize variables
    tomorrowRows = ""
    afterTomorrowRows = ""
    conditionMet = False

    ' Find the last row in column B
    lastRow = Cells(Rows.Count, 2).End(xlUp).Row

    ' Loop through each row in column B
    For currentRow = 1 To lastRow
        ' Get the date from the current row of column B
        Dim currentDate As Variant
        currentDate = Cells(currentRow, 2).Value

        ' Check if the value in column B is a valid date
        If IsDate(currentDate) Then
            ' Check if the date in column B is tomorrow or the day after tomorrow
            If currentDate = targetDateTomorrow Then
                ' Add the current row to the list of rows for tomorrow
                tomorrowRows = tomorrowRows & "Row " & currentRow & vbCrLf
                conditionMet = True
            ElseIf currentDate = targetDateDayAfterTomorrow Then
                ' Add the current row to the list of rows for the day after tomorrow
                afterTomorrowRows = afterTomorrowRows & "Row " & currentRow & vbCrLf
                conditionMet = True
            End If
        End If
    Next currentRow

    ' Check if any condition was met
    If conditionMet Then
        ' Prepare the message
        Dim message As String
        message = "The following dates are found:" & vbCrLf
        
        ' Add rows for tomorrow, if any
        If tomorrowRows <> "" Then
            message = message & vbCrLf & "Tomorrow:" & vbCrLf & tomorrowRows
        End If
        
        ' Add rows for the day after tomorrow, if any
        If afterTomorrowRows <> "" Then
            message = message & vbCrLf & "After Tomorrow:" & vbCrLf & afterTomorrowRows
        End If
        
        ' Display the message
        MsgBox message, vbInformation, "Date Check"
    Else
        ' If no dates were found, you can close Excel here
        Application.Quit
    End If
End Sub
